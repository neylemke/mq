\begin{questions}
\question[1.0] Um elétron de massa $m$ está confinado em uma esfera de 
raio $a$, isto é está submetido aoo potencial: $V(r)=0$ se %r<a$ e 
$V(r)=\infty$ se $r>a$.

\begin{itemize}
\item Escreva a solução da equação de Schrödinger independente do tempo
para a função $u(r)=rR(r)$, onde $\psi (r,\theta , \phi)=R(r)Y_{lm}(\theta , \phi)$
\item Determine a energia do estado fundamental.
\item Determine a autofunção associada.
\item Represente $|\psi|^2$ no plano $z=0$.
\end{itemize}

 
\end{questions}
