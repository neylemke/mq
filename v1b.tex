\begin{questions}
  \question[2.0] Uma partícula obedecendo a um potencial do tipo
  po\c{c}o infinito descrito por:

  \[
  V(x)= \left\{ \begin{array}{ll}
      0  & \mbox{se $0<x<a$} \\
      \infty & \mbox{se $0>x>a$ }
    \end{array}
  \right.
  \]

  Encontra-se inicialmente em um pacote de onda descrito por :
  \[
  \psi(x)= A\delta(x-a/4)-B\delta(x-3a/4)
  \]
  \begin{parts}
  \item Determine a condição em $A$ e $B$ para que $\psi$ esteja normalizada
e para que $\langle x\rangle=0$
  \item Determine qual é a probabilidade de encontrar a partícula no 
segundo estado excitado em $t=0$.  
  \end{parts}

\question[2.0] Considere uma partícula em um potencial como descrito na 
figura abaixo:
\begin{center}
\includegraphics[scale=0.25]{potential.png}
\end{center}
Assumindo que as partículas incidem sobre a barreira com energia cinética $E$ 
satisfazendo $V<E<U$ com direção e sentido dados pelo eixo $x$.  
\begin{parts}
  \item Escreva as equações que a função de onda $\psi$ e sua derivada  
devem obedecer em $x=b$.
\end{parts}
\end{questions}
