\begin{questions}

\question[2.0] Considere uma partícula em um poço infinito localizado
entre 0 e $a$, descrito
pela função de onda em $t=0$:
$$\psi(x,0)= A x (x-a/2) (x-a)$$ 

se $x>0$ e $x<a$ e zero nos demais casos. 
\begin{parts}
  \item Determine o valor da constante de normalização $A$
  \item Calcule a função de onda no espaço de momento. 
  \item Calcule o valor esperado da posição  em $t=0$.
  \item Qual é a probabilide em uma medida de energia encontrar o
    sistema no primeiro estado excitado. 
\end{parts}

\question[2.0] Considere a função:

$$\psi(x,t)=A e ^{-a (m x^2/\hbar +i t)}$$

\begin{parts}
  \item Determine $A$.
  \item Para qual função de energia potencial $V(x)$ $\psi$ satisfaz a
    equação de Schrödinger. 
\end{parts}





\question[2.0] Considere os autoestados coerentes do oscilador harmônico em
uma dimensão:

$$a^\dagger |\alpha\rangle=\alpha|\alpha\rangle$$

Calcule:

\begin{parts}
\item $\langle \alpha | x | \alpha \rangle$
\item $\langle \alpha | x^2 | \alpha \rangle$
\end{parts}

\question[2.0] Considere o vetor 
$$|\psi\rangle=\alpha \left(-1|10\rangle- 3 |1-1\rangle+2|11\rangle\right)$$
onde os vetores $|lm\rangle$ representam os 
autoestados de momento angular.  Calcule:
\begin{parts}
\item $\alpha$ para que o vetor esteja normalizado.
\item $\langle L^2 \rangle$
\item $\langle L_x \rangle$
\item qual é a probabilidade de ser medido $L_z=-\hbar$.
\end{parts}

\question[2.0] Calcule $\langle r \rangle$ e $\langle r ^2\rangle$  
para um átomo no estado fundamental do átomo de hidrogênio.  


\end{questions}
